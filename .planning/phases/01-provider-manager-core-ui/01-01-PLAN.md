    -- lua/ai/provider_manager/file_util.lua
    -- Safe file operations: atomic writes (.tmp → rename), Lua table parsing
    -- Addresses review concern: HIGH — file persistence without backup/atomic write
    
    local M = {}
    
    -- Write content to path via atomic .tmp → rename pattern
    function M.safe_write_file(path, content)
      local tmp_path = path .. ".tmp"
      
      -- Step 1: Write to .tmp file
      local ok, err = pcall(function()
        local lines = vim.split(content, "\n")
        vim.fn.writefile(lines, tmp_path)
      end)
      
      if not ok then
        vim.notify("Failed to write temp file: " .. tostring(err), vim.log.levels.ERROR)
        -- Clean up .tmp if it exists
        if vim.fn.filereadable(tmp_path) == 1 then
          pcall(vim.fn.delete, tmp_path)
        end
        return false, tostring(err)
      end
      
      -- Step 2: Rename .tmp → target (atomic on most filesystems)
      local ok2, err2 = pcall(function()
        -- vim.loop.fs_rename for atomic rename
        local uv = vim.loop or vim.uv
        if uv and uv.fs_rename then
          local result = uv.fs_rename(tmp_path, path)
          if result then
            error("fs_rename failed: " .. tostring(result))
          end
        else
          -- Fallback: delete old, copy new
          if vim.fn.filereadable(path) == 1 then
            pcall(vim.fn.delete, path)
          end
          pcall(vim.loop.fs_stat, path) -- ensure path is valid
          -- Use vim.fn.writefile directly as last resort
          local lines = vim.fn.readfile(tmp_path)
          vim.fn.writefile(lines, path)
        end
      end)
      
      if not ok2 then
        vim.notify("Failed to rename temp file: " .. tostring(err2), vim.log.levels.ERROR)
        return false, tostring(err2)
      end
      
      return true, nil
    end
    
    -- Read and parse a Lua file returning its table
    function M.read_lua_table(path)
      if vim.fn.filereadable(path) == 0 then
        return nil, "File not found: " .. path
      end
      local ok, result = pcall(dofile, path)
      if not ok then
        return nil, "Failed to parse Lua file: " .. tostring(result)
      end
      return result, nil
    end
    
    return M
    ```

    -- lua/ai/provider_manager/registry.lua
    -- CRUD operations with safe file persistence
    -- Addresses: review HIGH — delete line matching unreliable, file persistence unsafe
    
    local Providers = require("ai.providers")
    local Validator = require("ai.provider_manager.validator")
    local Keys = require("ai.keys")
    local FileUtil = require("ai.provider_manager.file_util")
    
    local M = {}
    
    ----------------------------------------------------------------------
    -- List providers (FIX: uses Providers.list() per review)
    ----------------------------------------------------------------------
    function M.list_providers()
      local result = {}
      local names = Providers.list()  -- returns array of names
      for _, name in ipairs(names) do
        local def = Providers.get(name)
        if def then
          local display = string.format("%s  —  %s  —  %s", name, def.endpoint, def.model or "unknown")
          table.insert(result, { name = name, display = display, endpoint = def.endpoint, model = def.model })
        end
      end
      return result
    end
    
    ----------------------------------------------------------------------
    -- Find provider block (addresses review: unreliable regex delete)
    ----------------------------------------------------------------------
    function M.find_provider_block(name)
      local path = vim.fn.stdpath("config") .. "/lua/ai/providers.lua"
      if vim.fn.filereadable(path) == 0 then
        return nil, nil, nil
      end
      local lines = vim.fn.readfile(path)
      local start_line = nil
      local end_line = nil
      
      for i, line in ipairs(lines) do
        if line:match("M%.register%(['\"]" .. name .. "['\"]") then
          start_line = i
        end
        if start_line and line:match("^%s*%}%s*%)\s*$") and i > start_line then
          end_line = i
          break
        end
      end
      
      if start_line and end_line then
        local content = {}
        for i = start_line, end_line do
          table.insert(content, lines[i])
        end
        return start_line, end_line, content
      end
      
      return nil, nil, nil
    end
    
    function M.find_provider_line(name)
      local start, _, _ = M.find_provider_block(name)
      return start or 1
    end
    
    ----------------------------------------------------------------------
    -- Add provider (opens providers.lua for manual config)
    ----------------------------------------------------------------------
    function M.add_provider(name)
      local valid, err = Validator.validate_provider_name(name)
      if not valid then
        vim.notify(err, vim.log.levels.ERROR)
        return false
      end
      local path = vim.fn.stdpath("config") .. "/lua/ai/providers.lua"
      vim.cmd("edit " .. path)
      local _, last_line, _ = M.find_provider_block(name)
      -- Jump to after last register call (generic: find last M.register end)
      local lines = vim.fn.readfile(path)
      local last_register_end = 0
      for i, line in ipairs(lines) do
        if line:match("M%.register%(") then
          last_register_end = i
        end
      end
      vim.api.nvim_win_set_cursor(0, { last_register_end + 2, 0 })
      vim.notify("Add provider config for: " .. name, vim.log.levels.INFO)
      return true
    end
    
    ----------------------------------------------------------------------
    -- Delete provider (FIX: block-aware parser + safe_write_file)
    ----------------------------------------------------------------------
    function M.delete_provider(name)
      if not Providers.get(name) then
        vim.notify("Provider not found: " .. name, vim.log.levels.ERROR)
        return false
      end
      
      -- Remove from in-memory Providers table
      Providers[name] = nil
      
      -- Clean up keys
      local keys = Keys.read()
      if keys and keys.providers and keys.providers[name] then
        keys.providers[name] = nil
        Keys.write(keys)
      end
      
      -- PERSIST: Remove M.register block from file
      local path = vim.fn.stdpath("config") .. "/lua/ai/providers.lua"
      if vim.fn.filereadable(path) == 0 then
        vim.notify("Deleted provider: " .. name .. " (file not found)", vim.log.levels.WARN)
        return true
      end
      
      local start, end_line, _ = M.find_provider_block(name)
      if not start then
        vim.notify("Could not find provider block in file", vim.log.levels.ERROR)
        return false
      end
      
      -- Read file, remove lines from start to end_line, write atomically
      local lines = vim.fn.readfile(path)
      local new_lines = {}
      for i, line in ipairs(lines) do
        if i < start or i > end_line then
          table.insert(new_lines, line)
        end
      end
      
      local content = table.concat(new_lines, "\n")
      local ok, err = FileUtil.safe_write_file(path, content)
      if not ok then
        vim.notify("Failed to persist deletion: " .. tostring(err), vim.log.levels.ERROR)
        return false
      end
      
      vim.notify("Deleted provider: " .. name, vim.log.levels.INFO)
      return true
    end
    
    return M
    ```
