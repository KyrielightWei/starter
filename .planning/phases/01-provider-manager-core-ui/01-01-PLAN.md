    ```lua
    local Providers = require("ai.providers")
    local Validator = require("ai.provider_manager.validator")
    local Keys = require("ai.keys")
    
    local M = {}
    
    -- Format providers for picker display (per D-01, UI-SPEC)
    -- FIX: Use Providers.list() API (per review HIGH concern) — NOT pairs(Providers)
    function M.list_providers()
      local result = {}
      -- Providers.list() returns array of provider names
      local names = Providers.list()
      for _, name in ipairs(names) do
        local def = Providers.get(name)
        if def then
          local display = string.format("%s  —  %s  —  %s", name, def.endpoint, def.model or "unknown")
          table.insert(result, { name = name, display = display, endpoint = def.endpoint, model = def.model })
        end
      end
      return result
    end
    
    -- Find line number of provider's M.register() call
    -- FIX: Use dynamic path via vim.fn.stdpath("config") (per review HIGH concern)
    function M.find_provider_line(name)
      local path = vim.fn.stdpath("config") .. "/lua/ai/providers.lua"
      if vim.fn.filereadable(path) == 0 then
        return 1
      end
      local lines = vim.fn.readfile(path)
      for i, line in ipairs(lines) do
        if line:match("M.register%(['\"]" .. name .. "['\"]") then
          return i
        end
      end
      return 1  -- Default to top of file
    end
    
    -- Add provider: opens providers.lua for manual config entry
    function M.add_provider(name)
      local valid, err = Validator.validate_provider_name(name)
      if not valid then
        vim.notify(err, vim.log.levels.ERROR)
        return false
      end
      -- Open providers.lua at end of register section for user to add
      local path = vim.fn.stdpath("config") .. "/lua/ai/providers.lua"
      vim.cmd("edit " .. path)
      -- Jump to last register line + 1
      local last_line = 0
      local lines = vim.fn.readfile(path)
      for i, line in ipairs(lines) do
        if line:match("M.register%(") then
          last_line = i
        end
      end
      vim.api.nvim_win_set_cursor(0, { last_line + 2, 0 })
      vim.notify("Add provider config for: " .. name, vim.log.levels.INFO)
      return true
    end
    
    -- Delete provider from registry AND persist to file
    -- FIX: File persistence (per review HIGH concern) — NOT just memory deletion
    -- Also cleanup keys (per qwen review MEDIUM concern)
    function M.delete_provider(name)
      if not Providers.get(name) then
        vim.notify("Provider not found: " .. name, vim.log.levels.ERROR)
        return false
      end
      -- Remove from Providers table (in-memory)
      Providers[name] = nil
      -- Remove provider's Keys entry
      local keys = Keys.read()
      if keys and keys[name] then
        keys[name] = nil
        Keys.write(keys)
      end
      -- PERSIST: Remove M.register(...) block from providers.lua file
      local path = vim.fn.stdpath("config") .. "/lua/ai/providers.lua"
      if vim.fn.filereadable(path) == 0 then
        vim.notify("Deleted provider: " .. name .. " (in-memory only, file not found)", vim.log.levels.WARN)
        return true
      end
      local lines = vim.fn.readfile(path)
      local new_lines = {}
      local skip = false
      for _, line in ipairs(lines) do
        if line:match("M%.register%(['\"]" .. name .. "['\"]") then
          skip = true
        elseif skip then
          if line:match("^%s*%}%s*%)\s*$") then
            -- End of register call: "})" closes the provider object
            skip = false
            -- Do NOT include this line (we removed the provider)
          end
          -- Skip all lines within the register block
        else
          table.insert(new_lines, line)
        end
      end
      if skip then
        vim.notify("Warning: Could not find end of register block for " .. name, vim.log.levels.WARN)
      end
      vim.fn.writefile(new_lines, path)
      vim.notify("Deleted provider: " .. name, vim.log.levels.INFO)
      return true
    end
    
    return M
    ```
