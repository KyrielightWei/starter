    ```lua
    -- lua/ai/provider_manager/picker.lua
    -- FZF-lua picker with two-step state machine and CRUD actions
    -- Addresses review concerns: keymap density (4 keys), CR semantics (state machine),
    -- auto-refresh, empty state guard, closure fix for async callbacks
    
    local M = {}
    
    local Registry = require("ai.provider_manager.registry")
    local Validator = require("ai.provider_manager.validator")
    
    -- State constants
    local STATE_PROVIDER = 1
    local STATE_MODEL = 2
    
    ----------------------------------------------------------------------
    -- Provider Step (Step 1)
    ----------------------------------------------------------------------
    function M.open()
      local ok, fzf = pcall(require, "fzf-lua.provided")
      if not ok then
        ok, fzf = pcall(require, "fzf-lua")
      end
      if not ok then
        vim.notify("fzf-lua not installed", vim.log.levels.ERROR)
        return
      end
      
      local providers = Registry.list_providers()
      
      local items = {}
      local name_map = {}
      for _, p in ipairs(providers) do
        table.insert(items, p.display)
        name_map[p.display] = p.name
      end
      
      -- Empty state guard (per review: no placeholder in items)
      if #items == 0 then
        vim.notify("No providers registered. Use <C-a> to add one.", vim.log.levels.WARN)
        return
      end
      
      fzf.fzf_exec(items, {
        prompt = "Providers> ",
        winopts = { width = 0.6, height = 0.5, border = "rounded" },
        actions = {
          -- <CR> Select provider → Step 2 (model picker)
          -- Addresses review HIGH: clear CR semantics per state
          ["default"] = function(selected)
            if not selected or #selected == 0 then return end
            local display = selected[1]
            local name = name_map[display]
            if not name then return end
            M._open_model_picker(name, fzf)
          end,
          
          -- <C-a> Add provider (addresses review: closure captures name)
          ["ctrl-a"] = function() M.add_provider_dialog() end,
          
          -- <C-d> Delete provider (addresses review: name captured before async)
          ["ctrl-d"] = function(selected)
            if not selected or #selected == 0 then return end
            local display = selected[1]
            local name = name_map[display]
            if not name then
              vim.notify("Provider not found", vim.log.levels.ERROR)
              return
            end
            M.delete_provider_dialog(name)
          end,
          
          -- <C-e> Edit provider
          ["ctrl-e"] = function(selected)
            if not selected or #selected == 0 then return end
            local display = selected[1]
            local name = name_map[display]
            if not name then return end
            M.edit_provider(name)
          end,
          
          -- <C-?> Help
          ["ctrl-/"] = function() M.show_help() end,
        },
        fzf_opts = {
          ["--header"] = "Actions: <CR>Select <C-a>Add <C-d>Delete <C-e>Edit <C-?>Help",
        },
      })
    end
    
    ----------------------------------------------------------------------
    -- Model Step (Step 2) — Separate picker, no CR conflict
    ----------------------------------------------------------------------
    function M._open_model_picker(provider_name, fzf)
      local def = Providers.get(provider_name)
      if not def then return end
      
      -- Try dynamic fetch, fallback to static_models
      local ok, Fetch = pcall(require, "ai.fetch_models")
      local models = {}
      if ok then
        local ok2, fetched = pcall(Fetch.fetch, provider_name)
        if ok2 and fetched and #fetched > 0 then
          models = fetched
        else
          models = def.static_models or {}
        end
      else
        models = def.static_models or {}
      end
      
      if #models == 0 then
        vim.notify("No models available for " .. provider_name, vim.log.levels.WARN)
        return
      end
      
      -- Current default model (move to top)
      local current_model = def.model
      local sorted = {}
      local seen = {}
      if current_model then
        table.insert(sorted, current_model)
        seen[current_model] = true
      end
      for _, m in ipairs(models) do
        if not seen[m] then
          table.insert(sorted, m)
          seen[m] = true
        end
      end
      
      fzf.fzf_exec(sorted, {
        prompt = string.format("Models (%s)> ", provider_name),
        winopts = { width = 0.5, height = 0.5, border = "rounded" },
        actions = {
          -- <CR> Select model → set as default, close (different from Step 1!)
          ["default"] = function(selected)
            if not selected or #selected == 0 then return end
            local model_id = selected[1]
            Registry.set_default_model(provider_name, model_id)
            vim.notify(string.format("Switched to %s / %s", provider_name, model_id), vim.log.levels.INFO)
            -- Auto-refresh provider picker
            vim.defer_fn(function() M.open() end, 50)
          end,
          ["ctrl-/"] = function() M.show_help("model") end,
        },
        fzf_opts = {
          ["--header"] = "Actions: <CR>Select Model <C-?>Help | <Esc> Back to providers",
        },
      })
    end
    
    ----------------------------------------------------------------------
    -- Dialog Functions (per review: closure fix, focus handling)
    ----------------------------------------------------------------------
    function M.add_provider_dialog()
      vim.ui.input({ prompt = "New provider name: " }, function(name)
        if not name or name == "" then return end
        local valid, err = Validator.validate_provider_name(name)
        if not valid then
          vim.notify(err, vim.log.levels.ERROR)
          return
        end
        Registry.add_provider(name)
        -- Auto-refresh picker after add
        vim.defer_fn(function() M.open() end, 50)
      end)
    end
    
    function M.delete_provider_dialog(name)
      vim.ui.input({ prompt = string.format("Delete provider '%s'? Type 'yes' to confirm: ", name) }, function(answer)
        if answer == "yes" then
          local ok = Registry.delete_provider(name)
          if ok then
            -- Auto-refresh picker after delete
            vim.defer_fn(function() M.open() end, 50)
          end
        end
      end)
    end
    
    function M.edit_provider(name)
      local line = Registry.find_provider_line(name)
      local path = vim.fn.stdpath("config") .. "/lua/ai/providers.lua"
      vim.cmd("edit " .. path)
      vim.api.nvim_win_set_cursor(0, { line, 0 })
      vim.notify("Editing provider: " .. name, vim.log.levels.INFO)
    end
    
    ----------------------------------------------------------------------
    -- Help Window
    ----------------------------------------------------------------------
    function M.show_help(step)
      step = step or "provider"
      local text = step == "provider" and [[
Provider Manager — Provider Step

  <CR>      Select provider → show models
  <C-a>     Add new provider
  <C-d>     Delete selected provider (confirmation required)
  <C-e>     Edit providers.lua directly
  <C-?>     Show this help

Press q to close
]] or [[
Provider Manager — Model Step

  <CR>      Select model → set as default
  <Esc>     Back to provider list
  <C-?>     Show this help

Press q to close
]]
      
      local buf = vim.api.nvim_create_buf(false, true)
      vim.api.nvim_buf_set_lines(buf, 0, -1, false, vim.split(text, "\n"))
      vim.api.nvim_buf_set_option(buf, "filetype", "help")
      
      local width = 55
      local height = 12
      local opts = {
        relative = "editor",
        width = width,
        height = height,
        col = (vim.o.columns - width) / 2,
        row = (vim.o.lines - height) / 2,
        style = "minimal",
        border = "rounded",
      }
      
      local win = vim.api.nvim_open_win(buf, true, opts)
      vim.keymap.set("n", "q", function()
        vim.api.nvim_win_close(win, true)
      end, { buffer = buf })
    end
    
    return M
    ```
