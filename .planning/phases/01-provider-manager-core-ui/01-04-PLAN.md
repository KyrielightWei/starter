    ```lua
    ----------------------------------------------------------------------
    -- Model Management (addresses review: boundary clarification)
    -- Keys store = user preferences (default model)
    -- Providers table = code definitions (static_models as fallback)
    ----------------------------------------------------------------------
    
    function M.list_models(provider_name)
      local def = Providers.get(provider_name)
      if not def then
        vim.notify("Provider not found: " .. provider_name, vim.log.levels.ERROR)
        return {}
      end
      
      -- Try dynamic fetch first, fallback to static_models
      local ok, Fetch = pcall(require, "ai.fetch_models")
      if ok then
        local ok2, models = pcall(Fetch.fetch, provider_name)
        if ok2 and models and #models > 0 then
          return models
        end
      end
      
      return def.static_models or {}
    end
    
    function M.set_default_model(provider_name, model_id)
      -- Read current keys config
      local config = Keys.read()
      if not config then
        vim.notify("Failed to read keys config", vim.log.levels.ERROR)
        return false
      end
      
      -- Ensure providers section exists
      if not config.providers then
        config.providers = {}
      end
      if not config.providers[provider_name] then
        config.providers[provider_name] = {}
      end
      
      -- Update default model in Keys (user preference)
      config.providers[provider_name].model = model_id
      
      -- Use safe file write if available
      local ok, FileUtil = pcall(require, "ai.provider_manager.file_util")
      if ok then
        local content = vim.fn.readfile(vim.fn.stdpath("config") .. "/lua/ai_keys.lua")
        -- Keys.write handles the serialization
        Keys.write(config)
        vim.notify(string.format("Set %s default model to: %s", provider_name, model_id), vim.log.levels.INFO)
      else
        Keys.write(config)
        vim.notify(string.format("Set %s default model to: %s", provider_name, model_id), vim.log.levels.INFO)
      end
      
      -- Also update in-memory Providers table for immediate effect
      local def = Providers.get(provider_name)
      if def then
        def.model = model_id
      end
      
      return true
    end
    
    function M.get_default_model(provider_name)
      -- Priority: Keys config > Providers.model > static_models[1]
      local config = Keys.read()
      if config and config.providers and config.providers[provider_name] and config.providers[provider_name].model then
        return config.providers[provider_name].model
      end
      
      local def = Providers.get(provider_name)
      if not def then return nil end
      
      return def.model or (def.static_models and def.static_models[1])
    end
    ```
