    1. **Add _edit_static_models function to picker.lua**:
       ```lua
       function M._edit_static_models(provider_name)
         local Registry = require("ai.provider_manager.registry")
         local ok, fzf = pcall(require, "fzf-lua")
         if not ok then return end
         
         local static_models = Registry.list_static_models(provider_name)
         local items = {}
         local id_map = {}
         
         for _, model_id in ipairs(static_models) do
           table.insert(items, model_id)
           id_map[model_id] = model_id
         end
         
         if #items == 0 then
           items = {"No static models. Ctrl-A to add."}
         end
         
         fzf.fzf_exec(items, {
           prompt = string.format("Static Models for %s> ", provider_name),
           fzf_opts = {
             ["--header"] = "Actions: <CR>Set Default <C-a>Add <C-d>Delete"
           },
           actions = {
             -- <CR> Set as default model
             ["default"] = function(selected)
               if not selected or #selected == 0 then return end
               local model_id = id_map[selected[1]]
               if not model_id then return end
               Registry.set_default_model(provider_name, model_id)
               -- Refresh picker
               vim.defer_fn(function() M.open() end, 100)
             end,
             
             -- <C-a> Add static model
             ["ctrl-a"] = function()
               vim.ui.input({ prompt = "Add static model: " }, function(input)
                 if not input or input == "" then return end
                 Registry.add_static_model(provider_name, input)
                 vim.defer_fn(function() M._edit_static_models(provider_name) end, 100)
               end)
             end,
             
             -- <C-d> Delete static model
             ["ctrl-d"] = function(selected)
               if not selected or #selected == 0 then return end
               local model_id = id_map[selected[1]]
               if not model_id then return end
               Registry.remove_static_model(provider_name, model_id)
               vim.notify("Removed: " .. model_id, vim.log.levels.INFO)
               -- Refresh editor
               vim.defer_fn(function() M._edit_static_models(provider_name) end, 100)
             end,
           },
         })
       end
       ```
