    Create lua/commit_picker/init.lua:

    ```lua
    -- lua/commit_picker/init.lua
    -- Commit Picker entry point

    local M = {}

    local Git = require("commit_picker.git")
    local Display = require("commit_picker.display")
    local Selection = require("commit_picker.selection")

    function M.setup(opts)
      opts = opts or {}

      -- Keymap: <leader>gd opens commit picker
      vim.keymap.set("n", "<leader>gd", function()
        M.open()
      end, { desc = "Commit Picker" })

      -- User command: :AICommitPicker
      vim.api.nvim_create_user_command("AICommitPicker", function()
        M.open()
      end, { desc = "Open commit picker" })

      return M
    end

    function M.open()
      local commits = Git.get_unpushed()

      if #commits == 0 then
        local ab = Git.get_ahead_behind()
        if ab.ahead == 0 then
          vim.notify("All commits are pushed to remote", vim.log.levels.INFO)
        else
          vim.notify("Could not fetch unpushed commits", vim.log.levels.WARN)
        end
        return
      end

      Display.show_picker(commits, {
        prompt = "Commit Picker (" .. #commits .. " commits)",
        on_select = function(sha)
          Selection.set_selected({ from = sha })
          vim.notify("Selected commit: " .. sha:sub(1, 7), vim.log.levels.INFO)
        end,
      })
    end

    return M
    ```

    Wire into the main AI module: modify `lua/ai/init.lua` to `require("commit_picker").setup()` during existing setup flow, if `commit_picker` is not already loaded there.
