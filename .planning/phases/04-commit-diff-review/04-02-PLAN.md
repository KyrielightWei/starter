   <action>
    Create lua/commit_picker/diff.lua:

    ```lua
    local M = {}

    -- Validate SHA format: 7-40 hex characters
    local function is_valid_sha(sha)
      return sha and sha:match("^%x%x%x%x%x%x%x+$")
    end

    function M.open_diff(shas)
      if not shas or #shas == 0 then
        vim.notify("请先选择 commit", vim.log.levels.WARN)
        return
      end

      -- Validate all SHAs before passing to vim.cmd (T-04-05)
      for _, sha in ipairs(shas) do
        if not is_valid_sha(sha) then
          vim.notify("无效的 SHA 格式: " .. sha, vim.log.levels.ERROR)
          return
        end
      end

      -- Check if diffview.nvim is available
      local ok = pcall(require, "diffview")
      if not ok then
        vim.notify("diffview.nvim 未配置，请手动运行 :DiffviewOpen", vim.log.levels.WARN)
        return
      end

      local range

      if #shas == 1 then
        -- Single commit: diff against parent (D-03, D-08)
        local sha = shas[1]
        range = sha .. "^.." .. sha
      elseif #shas >= 2 then
        -- Range diff between two commits (D-03, D-08)
        -- Use first two selected
        range = shas[1] .. ".." .. shas[2]
      else
        vim.notify("请选择 1 或 2 个 commit", vim.log.levels.WARN)
        return
      end

      -- Use DiffviewOpenEnhanced to preserve worktree support (D-09)
      -- This command is defined in lua/plugins/git.lua with dynamic git_cmd
      pcall(vim.cmd, "DiffviewOpenEnhanced " .. range)
    end

    return M
    ```

    Key decisions:
    - Single commit uses `sha^..sha` format — this is the standard git range for "show everything this commit introduced"
    - Two commits uses `sha1..sha2` — git diff between the two (inclusive of sha2, exclusive of sha1)
    - Uses `pcall(vim.cmd, ...)` for safety — diffview might have issues with invalid ranges
    - **Uses `DiffviewOpenEnhanced`** instead of `DiffviewOpen` to preserve worktree support defined in `lua/plugins/git.lua` (D-09)
    - Diffview opens in current tab by default (D-07) — no additional flags needed
    - Reuses existing diffview.nvim config from git.lua plugin spec (D-09) — no custom config passed
    - **SHA validation**: Validates each SHA as 7-40 hex chars before passing to vim.cmd (T-04-05 mitigation)
    - D-17 fallback: if `require("diffview")` fails, show warning with `:DiffviewOpen` suggestion
  </action>
