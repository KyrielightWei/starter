    ```lua
    -- lua/commit_picker/init.lua
    -- Commit Picker entry point

    local M = {}

    -- Lazy-load submodules with pcall to prevent total module failure
    local function get_modules()
      local ok, Git = pcall(require, "commit_picker.git")
      if not ok then return nil, "git" end
      local ok, Display = pcall(require, "commit_picker.display")
      if not ok then return nil, "display" end
      local ok, Selection = pcall(require, "commit_picker.selection")
      if not ok then return nil, "selection" end
      local ok, Diff = pcall(require, "commit_picker.diff")
      if not ok then return nil, "diff" end
      return { Git = Git, Display = Display, Selection = Selection, Diff = Diff }
    end

    function M.setup(opts)
      opts = opts or {}

      -- Keymap: <leader>kC opens commit picker (C = Commit, under AI <leader>k namespace, avoids collision with Phase 1's <leader>kp)
      vim.keymap.set("n", "<leader>kC", function()
        M.open()
      end, { desc = "Commit Picker" })

      -- User command: :AICommitPicker
      vim.api.nvim_create_user_command("AICommitPicker", function()
        M.open()
      end, { desc = "Open commit picker" })

      return M
    end

    function M.open()
      local mods, err = get_modules()
      if not mods then
        vim.notify("commit_picker." .. err .. " module failed to load", vim.log.levels.ERROR)
        return
      end

      local Git, Display, Selection, Diff = mods.Git, mods.Display, mods.Selection, mods.Diff

      -- Try unpushed commits first (D-02)
      local commits = Git.get_unpushed()

      -- Fallback: no unpushed commits -> show last 20 (D-15)
      if #commits == 0 then
        local ok, ab = pcall(function() return Git.get_ahead_behind() end)
        -- Always show info message when fallback triggers
        if ok and ab then
          vim.notify(string.format("没有未推送的提交 (ahead %d, behind %d)，显示最近 20 条提交", ab.ahead, ab.behind), vim.log.levels.INFO)
        else
          vim.notify("无法获取提交状态，显示最近 20 条提交", vim.log.levels.WARN)
        end
        commits = Git.get_commit_list(nil, "HEAD", { count = 20 })
        if #commits == 0 then
          vim.notify("没有找到提交", vim.log.levels.INFO)
          return
        end
      end

      Display.show_picker(commits, {
        on_select = function(selected_shas)
          -- Store selection
          Selection.set_selected(selected_shas)
          -- Open diff with selection (D-03, D-06, D-08)
          if #selected_shas > 0 then
            Diff.open_diff(selected_shas)
          end
        end,
      })
    end

    return M
    ```
